发布于 2019-05-31

好像是个傻逼题
我们考虑对于i来说j比k优
那么
dpi+(xi−xj)∗yi−ai>dpk+(xi−xk)∗yi−aidp_i+(x_i-x_j)*y_i-a_i>dp_k+(x_i-x_k)*y_i-a_idpi​+(xi​−xj​)∗yi​−ai​>dpk​+(xi​−xk​)∗yi​−ai​

也就是

dpj−xj∗yj>fk−xk∗yidp_j-x_j*y_j>f_k-x_k*y_idpj​−xj​∗yj​>fk​−xk​∗yi​

就变成了

$ \frac{(dp_j-dp_k)} {(x_j-x_k)}>y_i $

然后我们就可以维护左边了。同理维护一下右边。

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
struct rec{
    ll x,y,a;
    bool operator<(const rec &b)const {
        return x<b.x||(x==b.x&&y>b.y);
    }
};
int n;rec p[1000005];
ll dp[1000005];
ll q[1000005],l=0,r=0;
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%lld%lld%lld",&p[i].x,&p[i].y,&p[i].a);
    sort(p+1,p+1+n);
    for(int i=1;i<=n;i++){
        while (l<r&&(dp[q[l]]-dp[q[l+1]])<p[i].y*(p[q[l]].x-p[q[l+1]].x))l++;
        ll tmp = q[l];
        dp[i] = dp[tmp]+(p[i].x-p[tmp].x)*p[i].y-p[i].a;
        while (l<r&&1.0*(dp[i]-dp[q[r-1]])/(p[i].x-p[q[r-1]].x)>=1.0*(dp[q[r]]-dp[q[r-1]])/(p[q[r]].x-p[q[r-1]].x))r--;
        q[++r]=i;
    }
    ll x = *max_element(dp+1,dp+1+n);
    printf("%lld\n",x);
}
