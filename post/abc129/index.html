#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 2e3+3;
int n,m;
string s[N];
int l[N][N],r[N][N],u[N][N],d[N][N];
int main(){
    ios::sync_with_stdio(false);
    cin>>n>>m;
    for(int i=1;i<=n;i++){
        cin>>s[i];s[i]="#"+s[i]+'#';
    }
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
//            cout<<s[i][j];
            l[i][j]=(s[i][j]=='.'?l[i][j-1]+1:0);
//            cout<<l[i][j]<<endl;
        }
    }
    for(int i=1;i<=n;i++){
        for(int j=m;j>=1;j--){
            r[i][j]=(s[i][j]=='.'?r[i][j+1]+1:0);
        }
    }
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            u[i][j]=(s[i][j]=='.'?u[i-1][j]+1:0);
        }
    }
    for(int i=n;i>=1;i--){
        for(int j=m;j>=1;j--){
            d[i][j]=(s[i][j]=='.'?d[i+1][j]+1:0);
        }
    }
    int ans=0;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            if(l[i][j]+u[i][j]+r[i][j]+d[i][j]-3>ans){
                ans=l[i][j]+u[i][j]+r[i][j]+d[i][j]-3;
//                cout<<i<<' '<<j;
            }
//            ans = max(ans,);
        }
    }
    cout<<max(ans,0)<<endl;
}

    for(int i=0;i<n;i++){
