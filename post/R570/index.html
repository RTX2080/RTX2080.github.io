D：显然我们记录一下每个值的次数然后从大到小遍历一边就好了。

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 2e5+5;
int q,n,a[N],cnt[N],vis[N];
int main(){
    ios::sync_with_stdio(false);
    cin>>q;
    while (q--){
        cin>>n;
        for(int i=1;i<=n;i++){
            cin>>a[i];
            cnt[a[i]]++;
        }
        for(int i=1;i<=n;i++){
            vis[cnt[i]]++;
        }
        int pre=0,ans=0;
        for(int i=n;i>=1;i--){
            if(vis[i]){
                ans+=i;
                pre+=vis[i]-1;
            }else{
                if(pre)ans+=i,pre--;
            }
        }
        for(int i=1;i<=n;i++)a[i]=0,cnt[i]=0,vis[i]=0;
        cout<<ans<<endl;
    }
}

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 2e5+5;
int q,n,a[N],b[N],c[N],cnt[N],vis[N],pb[N];
vector<int> v[N];
int main(){
    ios::sync_with_stdio(false);
    cin>>q;
    while (q--){
        cin>>n;
        for(int i=1;i<=n;i++){
            cin>>a[i]>>b[i];
            cnt[a[i]]++;
            c[a[i]]+=b[i];//1
        }
        for(int i=1;i<=n;i++){
            vis[cnt[i]]++;
        }
        for(int i=1;i<=n;i++){
            if(!pb[a[i]]){
                v[cnt[a[i]]].push_back(c[a[i]]);
                pb[a[i]]=1;
//                cout<<cnt[a[i]]<<' '<<c[a[i]]<<endl;
            }
        }
        for(int i=1;i<=n;i++)sort(v[i].begin(),v[i].end()),reverse(v[i].begin(),v[i].end());
        int ans=0,ans2=0;
        priority_queue<int,vector<int>,less<int>> q;
        for(int i=n;i>=1;i--){
            if(vis[i]){
                for(int j=0;j<vis[i];j++)
                    q.push(v[i][j]);
            }
            if(!q.empty()){
                ans+=i,ans2+=min(i,q.top());
                q.pop();
            }
        }
        for(int i=1;i<=n;i++)a[i]=0,cnt[i]=0,vis[i]=0,v[i].clear(),b[i]=0,c[i]=0,pb[i]=0;
        cout<<ans<<' '<<ans2<<endl;
    }
}

H:

考虑dp[i][j]表示到第i个位置删掉j个字符的方案数。

那么我们要删去重复，其实也就是前面以 s[i] 结尾的子序列并且长度等于i-j

我直接n^3暴力找的。直接记录前缀下标就行。

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll n,k;
string s;
ll dp[105][105];
ll ans[105];
int main(){
    ios::sync_with_stdio(false);
    //cout<<1.0*(4.2*3+3.2+3.8*4.5+3.2*2+3.5*4.5+3.8*2.5)/(3+1+2+4.5+2.5+4.5);
    cin>>n>>k>>s;s="*"+s;
    dp[0][0]=1;//空串
    for(int i=1;i<=n;i++){
        dp[i][0]=1;
        for(int j=1;j<=i;j++){
            dp[i][j]=dp[i-1][j]+dp[i-1][j-1];
            for(int k=i-1;k>=1;k--){
                if(s[k]==s[i]) {
                    if(k-(i-j)>=0)
                        dp[i][j]-=dp[k-1][k-(i-j)];
                    break;
                }
            }
        }
    }
    ll ans=0;
    for(int i=0;i<=n&&k;i++){
        ans+=min(k,dp[n][i])*i;
        k-=min(k,dp[n][i]);
    }
    if(k)cout<<-1;
    else cout<<ans<<endl;
}

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 2e5+5;
int q,n,a[N];
int main(){
    ios::sync_with_stdio(false);
    cin>>q;
    while (q--){
        int mx1=0,mx2=0,mx3=0;
        cin>>n;
        for(int i=1;i<=n;i++){
            cin>>a[i];
        }
        mx1 = *max_element(a+1,a+1+n);
        vector<int> v;
        for(int i=1;i<=n;i++)if(mx1%a[i])v.push_back(a[i]);
        if(!v.empty()){
            mx2 = *max_element(v.begin(),v.end());
            vector<int> g;
            for(auto x:v)if(mx2%x)g.push_back(x);
            if(!g.empty()){
                mx3 = *max_element(g.begin(),g.end());
            }
        }
        int ans = mx1+mx2+mx3;
        bool f1=0,f2=0,f3=0;
        for(int i=1;i<=n;i++){
            if(mx1/2==a[i])f1=1;
            if(mx1/3==a[i])f2=1;
            if(mx1/5==a[i])f3=1;
        }
        if(f1&&f2&&f3)ans=max(ans,mx1/2+mx1/3+mx1/5);
        cout<<ans<<endl;
    }
}
