codeforces 1146H

发布于 2019-05-29

一开始没看懂题解甚至以为读错题了
和codeforces 13 D 有异曲同工之妙！
先来考虑不合法的情况叭
在森么样的情况下五个点才会不合法呢
其实对于一个五角星来说，如果两边的两个点缩到里面的话就gg了吧


也就是考虑这两种情况
凸四边形内有一个点，三角形内有两个点
分别对应了缩进去一个和两个
这两种情况都可以通过三角形内点的数目来得到
三角形内有多少个点bitset搞一下就行
那么对于第一种情况，我们考虑先拿出三角形和它里面的这个点
因为确保了三点不共线，所以剩下一个组成四边形的点随便拿
设三角形内有x个点，那么就是 x*(n-4)
看左边的图图发现会被count两次，除以2
很不幸的是这样对于第二种情况，我们也给算进去了，也就是算重复了
康康右边的图，本来只是一次不合法，你给人家算了两次。
所以我们还要再加上第二种的情况数，也就是C(n,2)

#include <bits/stdc++.h>
using namespace std;
typedef long long db;
int sign(db k){
    if (k>0) return 1; else if (k<0) return -1; return 0;
}
int cmp(db k1,db k2){return (k1-k2);}
int inmid(db k1,db k2,db k3){return (k1-k3)*(k2-k3)<=0;}// k3 在 [k1,k2] 内
struct point{
    db x,y;
    point operator + (const point &k1) const{return (point){k1.x+x,k1.y+y};}
    point operator - (const point &k1) const{return (point){x-k1.x,y-k1.y};}
    point operator * (db k1) const{return (point){x*k1,y*k1};}
    point operator / (db k1) const{return (point){x/k1,y/k1};}
    int operator == (const point &k1) const{return x==k1.x&&y==k1.y;}
};
int inmid(point k1,point k2,point k3){return inmid(k1.x,k2.x,k3.x)&&inmid(k1.y,k2.y,k3.y);}
db cross(point k1,point k2){return k1.x*k2.y-k1.y*k2.x;}
db dot(point k1,point k2){return k1.x*k2.x+k1.y*k2.y;}
db rad(point k1,point k2){return atan2(cross(k1,k2),dot(k1,k2));}
int n;
point p[333];
bitset<333> dp[333][333];
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%lld%lld",&p[i].x,&p[i].y);
    }
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            if(i==j)continue;
            for(int k=1;k<=n;k++){
                if(i==k||k==j)continue;
                if(cross(p[j]-p[i],p[k]-p[i])>0)dp[i][j].set(k);
            }
        }
    }
    db ans = 1ll*n*(n-1)*(n-2)*(n-3)*(n-4)/120;
    db cnt = 0;
    for(int i=1;i<=n;i++){
        for(int j=i+1;j<=n;j++){
            for(int k=i+1;k<=n;k++){
                if(j==k)continue;
                db s = (dp[i][j]&dp[j][k]&dp[k][i]).count();
                ans+=s*(s-1)/2;
                cnt+=s;
            }
        }
    }
    ans-=cnt*(n-4)/2;
    cout<<ans<<endl;
}
